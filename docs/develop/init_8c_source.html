186 #define handle_string_option(label, var) \
187  case label: \
188  ({ \
189  unsigned __i = 1; \
190  while(1) { \
191  if(strcmp(arg, param_to_text[key - OPT_FIRST][__i]) == 0) { \
192  var = __i; \
193  break; \
194  } \
195  if(!param_to_text[key - OPT_FIRST][++__i]) \
196  malformed_option_failure(); \
197  } \
198  }); \
199  break

203 #define parse_ullong_limits(low, high) \

207  __value = strtoull(arg, &__endptr, 10); \
208  if(!(*arg != '\0' && *__endptr == '\0' && (__value > low || __value == low) && __value <= high)) { \
209  malformed_option_failure(); \
210  } \
211  __value; \
212  })

214 static error_t parse_opt (int key, char *arg, struct argp_state *state)
215 {
216  // this is used in order to ensure that the user doesn't use duplicate options
217  static rootsim_bitmap scanned[bitmap_required_size(OPT_LAST - OPT_FIRST)];
218 
219  if(key >= OPT_FIRST && key < OPT_LAST){
220  if(bitmap_check(scanned, key - OPT_FIRST))
221  conflicting_option_failure("this option has already been specified");
222 
223  bitmap_set(scanned, key - OPT_FIRST);
224  }
225 
226  switch (key) {
227  case OPT_NP:
228  n_cores = parse_ullong_limits(1, UINT_MAX);
229  break;
230 
231  case OPT_NPRC:
232  n_prc_tot = parse_ullong_limits(1, UINT_MAX);
233  break;
234 
235  case OPT_OUTPUT_DIR:
236  rootsim_config.output_dir = arg;
237  break;
238 
239  handle_string_option(OPT_SCHEDULER, rootsim_config.scheduler);
240  handle_string_option(OPT_FULL, rootsim_config.snapshot);
241  handle_string_option(OPT_CKTRM_MODE, rootsim_config.check_termination_mode);
242  handle_string_option(OPT_VERBOSE, rootsim_config.verbose);
243  handle_string_option(OPT_STATS, rootsim_config.stats);
244  handle_string_option(OPT_LPS_DISTRIBUTION, rootsim_config.lps_distribution);
245 
246  case OPT_NPWD:
247  if (bitmap_check(scanned, OPT_P-OPT_FIRST)) {
248  conflicting_option_failure("I'm requested to run non piece-wise deterministically, but a checkpointing interval is set already.");
249  } else {
250  rootsim_config.checkpointing = STATE_SAVING_COPY;
251  }
252  break;
253 
254  case OPT_P:
255  if(bitmap_check(scanned, OPT_NPWD-OPT_FIRST)) {
256  conflicting_option_failure("Copy State Saving is selected, but I'm requested to set a checkpointing interval.");
257  } else {
258  rootsim_config.checkpointing = STATE_SAVING_PERIODIC;
259  rootsim_config.ckpt_period = parse_ullong_limits(1, 40);
260  // This is a micro optimization that makes the LogState function to avoid checking the checkpointing interval and keeping track of the logs taken
261  if(rootsim_config.ckpt_period == 1)
262  rootsim_config.checkpointing = STATE_SAVING_COPY;
263  }
264  break;
265 
266  case OPT_INC:
267  argp_failure(state, EXIT_FAILURE, ENOSYS, "incremental state saving is not supported in stable version yet...\nAborting");
268  break;
269 
270  case OPT_A:
271  argp_failure(state, EXIT_FAILURE, ENOSYS, "autonomic state saving is not supported in stable version yet...\nAborting");
272  break;
273 
274  case OPT_GVT:
275  rootsim_config.gvt_time_period = parse_ullong_limits(1, 10000);
276  break;
277 
278  case OPT_GVT_SNAPSHOT_CYCLES:
279  rootsim_config.gvt_snapshot_cycles = parse_ullong_limits(1, INT_MAX);
280  break;
281

284  break;

292  break;

309

320  rootsim_config.checkpointing = STATE_SAVING_PERIODIC;
321  rootsim_config.gvt_time_period = 1000;
322  rootsim_config.gvt_snapshot_cycles = 2;
323  rootsim_config.ckpt_period = 10;
324  rootsim_config.simulation_time = 0;
325  rootsim_config.deterministic_seed = false;
326  rootsim_config.set_seed = 0;
327  rootsim_config.serial = false;
328  rootsim_config.core_binding = true;
329 
330 #ifdef HAVE_PREEMPTION
331  rootsim_config.disable_preemption = false;
332 #endif
333  break;
334 
335  case ARGP_KEY_SUCCESS:
336 
337  // sanity checks
338  if(!rootsim_config.serial && !bitmap_check(scanned, OPT_NP - OPT_FIRST))
339  rootsim_error(true, "Number of cores was not provided \"--np\"\n");
340 
341  if(!bitmap_check(scanned, OPT_NPRC - OPT_FIRST))
342  rootsim_error(true, "Number of LPs was not provided \"--nprc\"\n");
343 
344  if(n_cores > get_cores())
345  rootsim_error(true, "Demanding %u cores, which are more than available (%d)\n", n_cores, get_cores());
346 
347  if(n_cores > MAX_THREADS_PER_KERNEL)
348  rootsim_error(true, "Too many threads, maximum supported number is %u\n", MAX_THREADS_PER_KERNEL);
349 
350  if(n_prc_tot > MAX_LPs)
351  rootsim_error(true, "Too many LPs, maximum supported number is %u\n", MAX_LPs);
352 
353  if(!rootsim_config.serial && n_prc_tot < n_cores)
354  rootsim_error(true, "Requested a simulation run with %u LPs and %u worker threads: the mapping is not possible\n", n_prc_tot, n_cores);
355 
356  print_config();
357 
358  break;
359  /* these functionalities are not needed
360  case ARGP_KEY_ARGS:
361  case ARGP_KEY_NO_ARGS:
362  case ARGP_KEY_SUCCESS:
363  case ARGP_KEY_END:
364  case ARGP_KEY_ARG:
365  case ARGP_KEY_ERROR:
366  */
367  default:
368  return ARGP_ERR_UNKNOWN;
369  }
370  return 0;
371 }

394 void SystemInit(int argc, char **argv)
395 {
396 #ifdef HAVE_MPI
397  mpi_init(&argc, &argv);
398 
399  if(n_ker > MAX_KERNELS){
400  rootsim_error(true, "Too many kernels, maximum supported number is %u\n", MAX_KERNELS);
401  }
402 #else
403  n_ker = 1;
404 #endif
405 
406  // Early initialization of ECS subsystem if needed
407 #ifdef HAVE_CROSS_STATE
408  ecs_init();
409 #endif
410 
411  // this retrieves the model's argp parser if declared by the developer
412  argp_child[0].argp = &model_argp;
413 
414  argp_parse (&argp, argc, argv, 0, NULL, NULL);
415 
416  // If we're going to run a serial simulation, configure the simulation to support it
417  if(rootsim_config.serial) {
418  ScheduleNewEvent = SerialScheduleNewEvent;
419  initialize_lps();
420  numerical_init();
421  statistics_init();
422  serial_init();
423  return;
424  } else {
425  ScheduleNewEvent = ParallelScheduleNewEvent;
426  }
427 
428  // Initialize ROOT-Sim subsystems.
429  // All init routines are executed serially (there is no notion of threads in there)
430  // and the order of invocation can matter!
431  base_init();
432  segment_init();
433  initialize_lps();
434  remote_memory_init();
435  statistics_init();
436  scheduler_init();
437  communication_init();
438  gvt_init();
439  numerical_init();
440 
441  // This call tells the simulation engine that the sequential initial simulation is complete
442  initialization_complete();
443 }
