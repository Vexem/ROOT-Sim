115 msg_t *get_msg_from_slab(void) {
116  msg_t *msg = NULL;
117  treiber *to_release;
118  treiber *to_release_nxt;
119 
120  // Unlink the whole Treiber stack and release all nodes.
121  // If at least one node is available, reuse it for the
122  // current allocation.
123  to_release = treiber_detach(msg_treiber[local_tid]);
124  while(to_release != NULL) {
125  if(unlikely(msg == NULL)) {
126  msg = to_release->data;
127  } else {
128  slab_free(&msg_slab[local_tid], to_release->data);
129  }
130  to_release_nxt = to_release->next;
131  rsfree(to_release);
132  to_release = to_release_nxt;
133  }
134 
135  if(likely(msg != NULL)) {
136  goto out;
137  }
138 
139  msg = (msg_t *)slab_alloc(&msg_slab[local_tid]);
140 
141  out:
142  bzero(msg, SLAB_MSG_SIZE);
143  msg->alloc_tid = local_tid;
144  return msg;
145 }

174 void ParallelScheduleNewEvent(unsigned int gid_receiver, simtime_t timestamp, unsigned int event_type, void *event_content, unsigned int event_size) {
175  msg_t *event;
176  GID_t receiver;
177 
178  switch_to_platform_mode();
179 
180  // Internally to the platform, the receiver is a GID, while models
181  // have no difference across GIDs and LIDs. We convert here the passed
182  // id to a GID.
183  set_gid(receiver, gid_receiver);
184 
185  // In Silent execution, we do not send again already sent messages
186  if(unlikely(LPS(current_lp)->state == LP_STATE_SILENT_EXEC)) {
187  return;
188  }
189 
190  // Check whether the destination LP is out of range
191  if(unlikely(receiver.id > n_prc_tot - 1)) { // It's unsigned, so no need to check whether it's < 0
192  rootsim_error(false, "Warning: the destination LP %u is out of range. The event has been ignored\n", receiver.id);
193  goto out;
194  }
195 
196  // Check if the associated timestamp is negative
197  if(unlikely(timestamp < lvt(current_lp))) {
198  rootsim_error(true, "LP %u is trying to generate an event (type %d) to %u in the past! (Current LVT = %f, generated event's timestamp = %f) Aborting...\n", current_lp, event_type, receiver.id, lvt(current_lp), timestamp);
199  }
200 
201  // Check if the event type is mapped to an internal control message
202  if(unlikely(event_type >= MIN_VALUE_CONTROL)) {
203  rootsim_error(true, "LP %u is generating an event with type %d which is a reserved type. Switch event type to a value less than %d. Aborting...\n", current_lp, event_type, MIN_VALUE_CONTROL);
204  }
205 
206 
207  // Copy all the information into the event structure
208  pack_msg(&event, LidToGid(current_lp), receiver, event_type, timestamp, lvt(current_lp), event_size, event_content);
209  event->mark = generate_mark(current_lp);
210 
211  if(unlikely(event->type == RENDEZVOUS_START)) {
212  event->rendezvous_mark = current_evt->rendezvous_mark;
213  printf("rendezvous_start mark=%llu\n",event->rendezvous_mark);
214  fflush(stdout);
215  }
216 
217  insert_outgoing_msg(event);
218 
219  out:
220  switch_to_application_mode();
221 }

235 void send_antimessages(LID_t lid, simtime_t after_simtime) {
236  msg_hdr_t *anti_msg, *anti_msg_prev;
237  msg_t *msg;
238 
239  if (unlikely(list_empty(LPS(lid)->queue_out)))
240  return;
241 
242  // Scan the output queue backwards, sending all required antimessages
243  anti_msg = list_tail(LPS(lid)->queue_out);
244  while(anti_msg != NULL && anti_msg->send_time > after_simtime) {
245  msg = get_msg_from_slab();
246  hdr_to_msg(anti_msg, msg);
247  msg->message_kind = negative;
248 
249  Send(msg);
250 
251  // Remove the already-sent antimessage from the output queue
252  anti_msg_prev = list_prev(anti_msg);
253  list_delete_by_content(LPS(lid)->queue_out, anti_msg);
254  msg_hdr_release(anti_msg);
255  anti_msg = anti_msg_prev;
256  }
257 }
