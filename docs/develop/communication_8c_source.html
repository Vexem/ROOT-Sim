137 void ParallelScheduleNewEvent(unsigned int gid_receiver, simtime_t timestamp, unsigned int event_type, void *event_content, unsigned int event_size)
138 {
139  msg_t *event;
140  GID_t receiver;
141 
142  switch_to_platform_mode();
143 
144  // Internally to the platform, the receiver is a GID, while models
145  // have no difference across GIDs and LIDs. We convert here the passed
146  // id to a GID.
147  set_gid(receiver, gid_receiver);
148 
149  // In Silent execution, we do not send again already sent messages
150  if (unlikely(current->state == LP_STATE_SILENT_EXEC)) {
151  return;
152  }
153  // Check whether the destination LP is out of range
154  if (unlikely(gid_receiver > n_prc_tot - 1)) { // It's unsigned, so no need to check whether it's < 0
155  rootsim_error(false, "Warning: the destination LP %u is out of range. The event has been ignored\n", gid_receiver);
156  goto out;
157  }
158  // Check if the associated timestamp is negative
159  if (unlikely(timestamp < lvt(current))) {
160  rootsim_error(true, "LP %u is trying to generate an event (type %d) to %u in the past! (Current LVT = %f, generated event's timestamp = %f) Aborting...\n",
161  current->gid, event_type, gid_receiver,
162  lvt(current), timestamp);
163  }
164  // Check if the event type is mapped to an internal control message
165  if (unlikely(event_type >= MIN_VALUE_CONTROL)) {
166  rootsim_error(true, "LP %u is generating an event with type %d which is a reserved type. Switch event type to a value less than %d. Aborting...\n",
167  current->gid, event_type, MIN_VALUE_CONTROL);
168  }
169 
170  // Copy all the information into the event structure
171  pack_msg(&event, current->gid, receiver, event_type, timestamp,
172  lvt(current), event_size, event_content);
173  event->mark = generate_mark(current);
174 
175  if (unlikely(event->type == RENDEZVOUS_START)) {
176  event->rendezvous_mark = current_evt->rendezvous_mark;
177  printf("rendezvous_start mark=%llu\n", event->rendezvous_mark);
178  fflush(stdout);
179  }
180 
181  insert_outgoing_msg(event);
182 
183  out:
184  switch_to_application_mode();
185 }

196 void send_antimessages(struct lp_struct *lp, simtime_t after_simtime)
197 {
198  msg_hdr_t *anti_msg, *anti_msg_prev;
199  msg_t *msg;
200 
201  if (unlikely(list_empty(lp->queue_out)))
202  return;
203 
204  // Scan the output queue backwards, sending all required antimessages
205  anti_msg = list_tail(lp->queue_out);
206  while (anti_msg != NULL && anti_msg->send_time > after_simtime) {
207  msg = get_msg_from_slab(which_slab_to_use(anti_msg->sender, anti_msg->receiver));
208  hdr_to_msg(anti_msg, msg);
209  msg->message_kind = negative;
210 
211  Send(msg);
212 
213  // Remove the already-sent antimessage from the output queue
214  anti_msg_prev = list_prev(anti_msg);
215  list_delete_by_content(lp->queue_out, anti_msg);
216  msg_hdr_release(anti_msg);
217  anti_msg = anti_msg_prev;
218  }
219 }

285 {
286  register unsigned int i = 0;

296
