115 /*
116  * Receive messages sent from remote kernels and
117  * destined to locally hosted LPs.
118  *
119  * This function will block until all the currently pending messages
120  * have been received and inserted into the bottom-half.
121  *
122  * This function is thread-safe.
123  */
124 void receive_remote_msgs(void)
125 {
126  int size;
127  msg_t *msg;
128  MPI_Status status;
129  MPI_Message mpi_msg;
130  int pending;
131  struct lp_struct *lp;
132  GID_t gid;
133 
134  if (!spin_trylock(&msgs_lock))
135  return;
136 
137  while (true) {
138  lock_mpi();
139  MPI_Improbe(MPI_ANY_SOURCE, MPI_ANY_TAG, msg_comm, &pending, &mpi_msg, &status);
140  unlock_mpi();
141 
142  if (!pending)
143  goto out;
144 
145  MPI_Get_count(&status, MPI_BYTE, &size);
146 
147  if (likely(MSG_PADDING + size <= SLAB_MSG_SIZE)) {
148  set_gid(gid, status.MPI_TAG);
149  lp = find_lp_by_gid(gid);
150  msg = get_msg_from_slab(lp);
151  } else {
152  msg = rsalloc(MSG_PADDING + size);
153  bzero(msg, MSG_PADDING);
154  }
155  /* - `pending_msgs` and `MPI_Recv` need to be in the same critical section.
156  * I could start an MPI_Recv with an empty incoming queue.
157  * - `MPI_Recv` and `insert_bottom_half` need to be in the same critical section.
158  * messages need to be inserted in arrival order into the BH
159  */
160 
161  // Receive the message
162  lock_mpi();
163  MPI_Mrecv(((char *)msg) + MSG_PADDING, size, MPI_BYTE, &mpi_msg, MPI_STATUS_IGNORE);
164  unlock_mpi();
165 
166  validate_msg(msg);
167  insert_bottom_half(msg);
168  }
169  out:
170  spin_unlock(&msgs_lock);
171 }

183 /*
184  * Accumulate termination acknoledgements from remote kernels and update the `terminated` counter
185  *
186  * This function can be called at any point of the simulation
187  * but it will be effective only after that broadcast_termination() have been called locally.
188  *
189  * This function is thread-safe
190  */
191 void collect_termination(void)
192 {
193  int res;
194  unsigned int tdata;
195 
196  if (terminated == 0 || !spin_trylock(&msgs_fini))
197  return;
198 
199  while (pending_msgs(MSG_FINI)) {
200  lock_mpi();
201  res =
202  MPI_Recv(&tdata, 1, MPI_UNSIGNED, MPI_ANY_SOURCE, MSG_FINI, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
203  unlock_mpi();
204  if (unlikely(res != 0)) {
205  rootsim_error(true, "MPI_Recv did not complete correctly");
206  return;
207  }
208  terminated++;
209  }
210  spin_unlock(&msgs_fini);
211 }

213 /* Notify all the kernels about local termination
214  *
215  * This function can be called multiple time,
216  * but the actual broadcasting operation will be executed only
217  * on the first call.
218  *
219  * Multithread infos:
220  * This function is not thread safe and should be used
221  * by only one thread at time.
222  * This function can be used concurrently with other
223  * MPI communication function.
224  */
225 void broadcast_termination(void)
226 {
227  unsigned int i;
228  lock_mpi();
229  for (i = 0; i < n_ker; i++) {
230  if (i == kid)
231  continue;
232  MPI_Isend(&i, 1, MPI_UNSIGNED, i, MSG_FINI, MPI_COMM_WORLD, &termination_reqs[i]);
233  }
234  terminated++;
235  unlock_mpi();
236 }
