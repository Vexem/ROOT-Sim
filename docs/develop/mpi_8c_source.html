132 void receive_remote_msgs(void)
133 {
134  int size;
135  msg_t *msg;
136  MPI_Status status;
137  MPI_Message mpi_msg;
138  int pending;
139  struct lp_struct *lp;
140  GID_t gid;
141 
142  if (!spin_trylock(&msgs_lock))
143  return;
144 
145  while (true) {
146  lock_mpi();
147  MPI_Improbe(MPI_ANY_SOURCE, MPI_ANY_TAG, msg_comm, &pending, &mpi_msg, &status);
148  unlock_mpi();
149 
150  if (!pending)
151  goto out;
152 
153  MPI_Get_count(&status, MPI_BYTE, &size);
154 
155  if (likely(MSG_PADDING + size <= SLAB_MSG_SIZE)) {
156  set_gid(gid, status.MPI_TAG);
157  lp = find_lp_by_gid(gid);
158  msg = get_msg_from_slab(lp);
159  } else {
160  msg = rsalloc(MSG_PADDING + size);
161  bzero(msg, MSG_PADDING);
162  }
163  /* - `pending_msgs` and `MPI_Recv` need to be in the same critical section.
164  * I could start an MPI_Recv with an empty incoming queue.
165  * - `MPI_Recv` and `insert_bottom_half` need to be in the same critical section.
166  * messages need to be inserted in arrival order into the BH
167  */
168 
169  // Receive the message
170  lock_mpi();
171  MPI_Mrecv(((char *)msg) + MSG_PADDING, size, MPI_BYTE, &mpi_msg, MPI_STATUS_IGNORE);
172  unlock_mpi();
173 
174  validate_msg(msg);
175  insert_bottom_half(msg);
176  }
177  out:
178  spin_unlock(&msgs_lock);
179 }
