522 static astar_t a_star(unsigned int *a_star_bitmap, int topology, unsigned int current_cell, unsigned int dest, unsigned int step, obstacles_t *obstacles) {
523  unsigned int i;
524  unsigned int tentative_cell;
525 
526  astar_t states[8] = { [0 ... 7] = { UINT_MAX, NULL } };
527 
528  // TODO: again, we need a generic bitmap type in ROOT-Sim
529  unsigned int tentative_a_star_bitmap[n_prc_tot / NUM_CHUNKS_PER_BLOCK + 1];
530 
531  // Is this the target?
532  if (current_cell == dest) {
533  states[0].list = rsalloc(sizeof(unsigned int) * step);
534  states[0].list[step - 1] = current_cell;
535  states[0].num_steps = step;
536  return states[0];
537  }
538 
539  // We have visited this cell
540  SET_BIT(a_star_bitmap, current_cell);
541 
542  // Scan all possible neighbours depending on the actual topology
543  for(i = 0; i < (topology == TOPOLOGY_SQUARE ? 4 : 8); i++) {
544  tentative_cell = GetReceiver(TOPOLOGY_HEXAGON, i);
545 
546  // Try all reachable unvisited cells
547  if(tentative_cell == INVALID_DIRECTION || CHECK_BIT(a_star_bitmap, tentative_cell) || CHECK_BIT(obstacles->grid, tentative_cell))
548  continue;
549 
550  // Get closer to the destination
551  memcpy(tentative_a_star_bitmap, a_star_bitmap, sizeof(unsigned int) * (n_prc_tot / NUM_CHUNKS_PER_BLOCK + 1));
552  states[i] = a_star(tentative_a_star_bitmap, topology, tentative_cell, dest, step + 1, obstacles);
553  }
554 
555  // Pick the state with the minimum distance
556  qsort(states, (topology == TOPOLOGY_SQUARE ? 4 : 8), sizeof(astar_t), compare_astar);
557 
558  // Free unneeded states
559  for(i = 1; i < (topology == TOPOLOGY_SQUARE ? 4 : 8); i++) {
560  if(states[i].list != NULL) {
561  rsfree(states[i].list);
562  }
563  }
564 
565  // Register me in the 0-th state's list
566  states[0].list[step - 1] = current_cell;
567 
568  return states[0];
569 }
