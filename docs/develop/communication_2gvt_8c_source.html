346 void flush_white_msg_recv(void)
347 {
348  // santy check: Exactly the expected number of white
349  // messages should have been arrived in the last GVT round
350  if (atomic_read(white_msg_recv) != expected_white_msg) {
351  rootsim_error(true, "unexpected number of white messages received in the last GVT round: [expected: %d, received: %d]\n",
352  expected_white_msg, atomic_read(white_msg_recv));
353  }
354  // prepare the white msg counter for the next GVT round
