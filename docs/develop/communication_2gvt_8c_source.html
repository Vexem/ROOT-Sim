173 /*
174  * Join the white message reduction collective operation.
175  *
176  * All kernels will share eachother the number of white message sent
177  * during the last white phase.
178  *
179  * This is an asyncronous function and the actual completion of the collective
180  * reduction can be tested through the `white_msg_redux_completed()` function.
181  *
182  * After the collective reduction will be terminated `expected_white_msg`
183  * will hold the number of white message sent by all the other kernel to this during
184  * the last white phase.
185  */
186 void join_white_msg_redux(void) {
187  unsigned int i;
188  for(i = 0; i < n_ker; i++) {
189  white_msg_sent_buff[i] = atomic_read(&white_msg_sent[i]);
190  }
191 
192  lock_mpi();
193  MPI_Ireduce_scatter_block(white_msg_sent_buff, &expected_white_msg, 1, MPI_INT, MPI_SUM, white_count_comm, &white_count_req);
194  unlock_mpi();
195 }
