113 
114  if (unlikely(self == NULL || self->size < size)) {
115  rootsim_error(true,"size is %u < %u\n",self->size,size);
116  return -1;
117  }
118  size = POWEROF2(size);
119 
120  unsigned idx = 0;
121  if (unlikely(self->longest[idx] < size)) {
122  rootsim_error(true,"self->longest %u < %u\n",self->longest[idx],size);
123  return -1;
124  }
125 
126  /* search recursively for the child */
127  unsigned node_size = 0;
128  for (node_size = self->size; node_size != size; node_size >>= 1) {
