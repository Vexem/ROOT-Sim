335 void distribute_lps_on_kernels(void) {
336  register unsigned int i = 0;
337  unsigned int j;
338  unsigned int buf1;
339  int offset;
340  int block_leftover;
341 
342  // Sanity check on number of LPs
343  if(n_prc_tot < n_ker)
344  rootsim_error(true, "Unable to allocate %d logical processes on %d kernels: must have at least %d LPs\n", n_prc_tot, n_ker, n_ker);
345 
346  kernel = (unsigned int *)rsalloc(sizeof(unsigned int) * n_prc_tot);
347 
348 
349  switch (rootsim_config.lps_distribution) {
350 
351  case LP_DISTRIBUTION_BLOCK:
352  buf1 = (n_prc_tot / n_ker);
353  block_leftover = n_prc_tot - buf1 * n_ker;
354 
355  // It is a hack to bypass the first check that set offset to 0
356  if (block_leftover > 0)
357  buf1++;
358 
359  offset = 0;
360  while (i < n_prc_tot) {
361  j = 0;
362  while (j < buf1) {
363  kernel[i] = offset;
364  i++;
365  j++;
366  }
367  offset++;
368  block_leftover--;
369  if (block_leftover == 0)
370  buf1--;
371  }
372  break;
373 
374  case LP_DISTRIBUTION_CIRCULAR:
375  for (i = 0; i < n_prc_tot; i++) {
376  kernel[i] = i % n_ker;
377  }
378  break;
379  }
380 }
