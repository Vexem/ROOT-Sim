74 void ecs_secondary(void) {
75 
76  GID_t target_gid;
77  // target_address is filled by the ROOT-Sim fault handler at kernel level before triggering the signal
78  long long target_address = fault_info.target_address;
79  unsigned char *faulting_insn = (unsigned char *)fault_info.rip;
80  long long span;
81  insn_info_x86 insn_disasm;
82  unsigned long i=0;
83  msg_t *control_msg;
84  ecs_page_request_t page_req;
85 
86  // Disassemble the faulting instruction to get necessary information
87  x86_disassemble_instruction(faulting_insn, &i, &insn_disasm, DATA_64 | ADDR_64);
88 
89  if(!IS_STRING(&insn_disasm)) {
90  span = insn_disasm.span;
91  } else {
92  span = insn_disasm.span * fault_info.rcx;
93  }
94 
95  // Compute the starting page address and the page count to get the lease
96  page_req.write_mode = IS_MEMWR(&insn_disasm);
97  page_req.base_address = (void *)(target_address & (~((long long)PAGE_SIZE-1)));
98  page_req.count = ((target_address + span) & (~((long long)PAGE_SIZE-1)))/PAGE_SIZE - (long long)page_req.base_address/PAGE_SIZE + 1;
99 
100 // printf("ECS Page Fault: LP %d accessing %d pages from %p on LP %lu in %s mode\n", current_lp, page_req.count, (void *)page_req.base_address, fault_info.target_gid, (page_req.write_mode ? "write" : "read"));
101  fflush(stdout);
102 
103  // Send the page lease request control message. This is not incorporated into the input queue at the receiver
104  // so we do not place it into the output queue
105  target_gid.id = fault_info.target_gid;
106  pack_msg(&control_msg, LidToGid(current_lp), target_gid, RENDEZVOUS_GET_PAGE, current_lvt, current_lvt, sizeof(page_req), &page_req);
107  control_msg->rendezvous_mark = current_evt->rendezvous_mark;
108  Send(control_msg);
109 
110  LPS(current_lp)->state = LP_STATE_WAIT_FOR_DATA;
111 
112  // Give back control to the simulation kernel's user-level thread
113  long_jmp(&kernel_context, kernel_context.rax);
114 }

116 void ecs_initiate(void) {
117  msg_t *control_msg;
118  msg_hdr_t *msg_hdr;
119 
120  GID_t target_gid;
121  // Generate a unique mark for this ECS
122  current_evt->rendezvous_mark = generate_mark(current_lp);
123  LPS(current_lp)->wait_on_rendezvous = current_evt->rendezvous_mark;
124 
125  // Prepare the control message to synchronize the two LPs
126  target_gid.id = fault_info.target_gid;
127  pack_msg(&control_msg, LidToGid(current_lp), target_gid, RENDEZVOUS_START, current_lvt, current_lvt, 0, NULL);
128  control_msg->rendezvous_mark = current_evt->rendezvous_mark;
129  control_msg->mark = generate_mark(current_lp);
130 
131  // This message must be stored in the output queue as well, in case this LP rollbacks
132  msg_hdr = get_msg_hdr_from_slab();
133  msg_to_hdr(msg_hdr, control_msg);
134  list_insert(LPS(current_lp)->queue_out, send_time, msg_hdr);
135 
136  // Block the execution of this LP
137  LPS(current_lp)->state = LP_STATE_WAIT_FOR_SYNCH;
138  LPS(current_lp)->wait_on_object = fault_info.target_gid;
139 
140  // Store which LP we are waiting for synchronization.
141  LPS(current_lp)->ECS_index++;
142  LPS(current_lp)->ECS_synch_table[LPS(current_lp)->ECS_index] = target_gid;
143  Send(control_msg);
144 
145  // Give back control to the simulation kernel's user-level thread
146  long_jmp(&kernel_context, kernel_context.rax);
147 }

151 void ECS(void) {
152  // ECS cannot happen in silent execution, as we take a log after the completion
153  // of an event which involves one or multiple ecs
154  if(unlikely(LPS(current_lp)->state == LP_STATE_SILENT_EXEC)) {
155  rootsim_error(true,"----ERROR---- ECS in Silent Execution LP[%d] Hit:%llu Timestamp:%f\n",
156  current_lp, fault_info.target_gid, current_lvt);
157  }
158 
159  // Sanity check: we cannot run an ECS with an old mark after a rollback
160  if(LPS(current_lp)->wait_on_rendezvous != 0 && LPS(current_lp)->wait_on_rendezvous != current_evt->rendezvous_mark) {
161  printf("muori male\n");
162  fflush(stdout);
163  abort();
164  }
165 
166 // printf("Entro nell'ECS handler per un fault di tipo %d\n", fault_info.fault_type);
167 
168  switch(fault_info.fault_type) {
169 
170  case ECS_MAJOR_FAULT:
171  ecs_initiate();
172  break;
173 
174  case ECS_MINOR_FAULT:
175  // TODO: gestire qui la read/write list!!!!
176  ecs_secondary();
177  break;
178 
179  case ECS_CHANGE_PAGE_PRIVILEGE:
180  // TODO: gestire qui la write list!!!!
181  lp_alloc_schedule(); // We moved to the original view in the kernel module: we do not unschedule the LP here
182  break;
183 
184  default:
185  rootsim_error(true, "%s:%d: Impossible condition! Aborting...\n", __FILE__, __LINE__);
186  return;
187  }
188 }
