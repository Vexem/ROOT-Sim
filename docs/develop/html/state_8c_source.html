146 unsigned int silent_execution(LID_t lid, void *state_buffer, msg_t *evt, msg_t *final_evt) {
147  unsigned int events = 0;
148  unsigned short int old_state;
149 
150  // current state can be either idle READY, BLOCKED or ROLLBACK, so we save it and then put it back in place
151  old_state = LPS(lid)->state;
152  LPS(lid)->state = LP_STATE_SILENT_EXEC;
153 
154  // This is true if the restored state was taken exactly after the new bound
155  if(evt == final_evt)
156  goto out;
157 
158  evt = list_next(evt);
159  final_evt = list_next(final_evt);
160 
161  // Reprocess events. Outgoing messages are explicitly discarded, as this part of
162  // the simulation has been already executed at least once
163  while(evt != NULL && evt != final_evt) {
164 
165  if(unlikely(!reprocess_control_msg(evt))) {
166  evt = list_next(evt);
167  continue;
168  }
169 
170  events++;
171  activate_LP(lid, evt->timestamp, evt, state_buffer);
172  evt = list_next(evt);
173  }
174 
175 out:
176  LPS(lid)->state = old_state;
177  return events;
178 }

193 void rollback(LID_t lid) {
194  state_t *restore_state, *s;
195  msg_t *last_correct_event;
196  msg_t *last_restored_event;
197  unsigned int reprocessed_events;
198 
199 
200  // Sanity check
201  if(unlikely(LPS(lid)->state != LP_STATE_ROLLBACK)) {
202  rootsim_error(false, "I'm asked to roll back LP %d's execution, but rollback_bound is not set. Ignoring...\n", LidToGid(lid));
203  return;
204  }
205 
206 
207  // Discard any possible execution state related to a blocked execution
208  memcpy(&LPS(lid)->context, &LPS(lid)->default_context, sizeof(LP_context_t));
209 
210  statistics_post_data(lid, STAT_ROLLBACK, 1.0);
211 
212  last_correct_event = LPS(lid)->bound;
213  // Send antimessages
214  send_antimessages(lid, last_correct_event->timestamp);
215 
216  // Find the state to be restored, and prune the wrongly computed states
217  restore_state = list_tail(LPS(lid)->queue_states);
218  while (restore_state != NULL && restore_state->lvt > last_correct_event->timestamp) { // It's > rather than >= because we have already taken into account simultaneous events
219  s = restore_state;
220  restore_state = list_prev(restore_state);
221  log_delete(s->log);
222  #ifndef NDEBUG
223  s->last_event = (void *)0xBABEBEEF;
224  #endif
225  list_delete_by_content(LPS(lid)->queue_states, s);
226  }
227  // Restore the simulation state and correct the state base pointer
228  RestoreState(lid, restore_state);
229 
230  last_restored_event = restore_state->last_event;
231  reprocessed_events = silent_execution(lid, LPS(lid)->current_base_pointer, last_restored_event, last_correct_event);
232  statistics_post_data(lid, STAT_SILENT, (double)reprocessed_events);
233 
234  // TODO: silent execution resets the LP state to the previous
235  // value, so it should be the last function to be called within rollback()
236  // Control messages must be rolled back as well
237  rollback_control_message(lid, last_correct_event->timestamp);
238 }
