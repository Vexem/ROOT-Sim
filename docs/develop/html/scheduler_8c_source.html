406 void schedule(void) {
407 
408  LID_t lid;
409  msg_t *event;
410  void *state;
411 
412  #ifdef HAVE_CROSS_STATE
413  bool resume_execution = false;
414  #endif
415 
416  // Find next LP to be executed, depending on the chosen scheduler
417  switch (rootsim_config.scheduler) {
418 
419  case SCHEDULER_STF:
420  lid = smallest_timestamp_first();
421  break;
422 
423  default:
424  rootsim_error(true, "unrecognized scheduler!");
425  }
426 
427  // No logical process found with events to be processed
428  if (lid_equals(lid, idle_process)) {
429  statistics_post_data(lid, STAT_IDLE_CYCLES, 1.0);
430  return;
431  }
432 
433 // if(lid == 1 && LPS[lid]->state != LP_STATE_READY)
434 // printf("state of lid 1 is %d\n",LPS[lid]->state);
435 
436  // If we have to rollback
437  if(LPS(lid)->state == LP_STATE_ROLLBACK) {
438  rollback(lid);
439  LPS(lid)->state = LP_STATE_READY;
440  send_outgoing_msgs(lid);
441  return;
442  }
443 
444  if(!is_blocked_state(LPS(lid)->state) && LPS(lid)->state != LP_STATE_READY_FOR_SYNCH){
445  event = advance_to_next_event(lid);
446  }
447  else {
448  event = LPS(lid)->bound;
449  }
450 
451 
452  // Sanity check: if we get here, it means that lid is a LP which has
453  // at least one event to be executed. If advance_to_next_event() returns
454  // NULL, it means that lid has no events to be executed. This is
455  // a critical condition and we abort.
456  if(unlikely(event == NULL)) {
457  rootsim_error(true, "Critical condition: LP %d seems to have events to be processed, but I cannot find them. Aborting...\n", lid);
458  }
459 
460  if(unlikely(!process_control_msg(event))) {
461  return;
462  }
463 
464  state = LPS(lid)->current_base_pointer;
465 
466 
467 #ifdef HAVE_CROSS_STATE
468  // In case we are resuming an interrupted execution, we keep track of this.
469  // If at the end of the scheduling the LP is not blocked, we can unblock all the remote objects
470  if(is_blocked_state(LPS(lid)->state) || LPS(lid)->state == LP_STATE_READY_FOR_SYNCH) {
471  resume_execution = true;
472  }
473 #endif
474 
475 
476  // Schedule the LP user-level thread
477  if(LPS(lid)->state == LP_STATE_READY_FOR_SYNCH)
478  LPS(lid)->state = LP_STATE_RUNNING_ECS;
479  else
480  LPS(lid)->state = LP_STATE_RUNNING;
481 
482  activate_LP(lid, lvt(lid), event, state);
483 
484  if(!is_blocked_state(LPS(lid)->state)) {
485  LPS(lid)->state = LP_STATE_READY;
486  send_outgoing_msgs(lid);
487  }
488 
489 #ifdef HAVE_CROSS_STATE
490  if(resume_execution && !is_blocked_state(LPS(lid)->state)) {
491  //printf("ECS event is finished mark %llu !!!\n", LPS(lid)->wait_on_rendezvous);
492  fflush(stdout);
493  unblock_synchronized_objects(lid);
494 
495  // This is to avoid domino effect when relying on rendezvous messages
496  force_LP_checkpoint(lid);
497  }
498 #endif
499 
500  // Log the state, if needed
501  LogState(lid);
502 }
