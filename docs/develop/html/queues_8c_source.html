148 void process_bottom_halves(void) {
149  unsigned int i;
150  LID_t lid_receiver;
151  LP_State *receiver;
152 
153  msg_t *msg_to_process;
154  msg_t *matched_msg;
155 
156  for(i = 0; i < n_prc_per_thread; i++) {
157 
158  while((msg_to_process = get_msg(LPS_bound(i)->bottom_halves)) != NULL) {
159  lid_receiver = GidToLid(msg_to_process->receiver);
160  receiver = LPS(lid_receiver);
161 
162  // Sanity check
163  if(unlikely(msg_to_process->timestamp < get_last_gvt()))
164  rootsim_error(true, "The impossible happened: I'm receiving a message before the GVT\n");
165 
166  // Handle control messages
167  if(unlikely(!receive_control_msg(msg_to_process))) {
168  msg_release(msg_to_process);
169  continue;
170  }
171 
172  switch (msg_to_process->message_kind) {
173 
174  // It's an antimessage
175  case negative:
176 
177  statistics_post_data(lid_receiver, STAT_ANTIMESSAGE, 1.0);
178 
179  // Find the message matching the antimessage
180  matched_msg = list_tail(receiver->queue_in);
181  while(matched_msg != NULL && matched_msg->mark != msg_to_process->mark) {
182  matched_msg = list_prev(matched_msg);
183  }
184 
185  // Sanity check
186  if(unlikely(matched_msg == NULL)) {
187  rootsim_error(false, "LP %d Received an antimessage, but no such mark has been found!\n", lid_to_int(lid_receiver));
188  dump_msg_content(msg_to_process);
189  rootsim_error(true, "Aborting...\n");
190  }
191 
192  // If the matched message is in the past, we have to rollback
193  if(matched_msg->timestamp <= lvt(lid_receiver)) {
194 
195  receiver->bound = list_prev(matched_msg);
196  while((receiver->bound != NULL) && D_EQUAL(receiver->bound->timestamp, msg_to_process->timestamp)) {
197  receiver->bound = list_prev(receiver->bound);
198  }
199 
200  receiver->state = LP_STATE_ROLLBACK;
201 
202  }
203 
204  #ifdef HAVE_MPI
205  register_incoming_msg(msg_to_process);
206  #endif
207 
208  // Delete the matched message
209  list_delete_by_content(receiver->queue_in, matched_msg);
210  msg_release(matched_msg);
211 
212  break;
213 
214  // It's a positive message
215  case positive:
216 
217  // A positive message is directly placed in the queue
218  list_insert(receiver->queue_in, timestamp, msg_to_process);
219 
220  // Check if we've just inserted an out-of-order event.
221  // Here we check for a strictly minor timestamp since
222  // the queue is FIFO for same-timestamp events. Therefore,
223  // A contemporaneous event does not cause a causal violation.
224  if(msg_to_process->timestamp < lvt(lid_receiver)) {
225 
226  receiver->bound = list_prev(msg_to_process);
227  while((receiver->bound != NULL) && D_EQUAL(receiver->bound->timestamp, msg_to_process->timestamp)) {
228  receiver->bound = list_prev(receiver->bound);
229  }
230 
231  receiver->state = LP_STATE_ROLLBACK;
232  }
233 #ifdef HAVE_MPI
234  register_incoming_msg(msg_to_process);
235 #endif
236  break;
237 
238  // It's a control message
239  case control:
240 
241  // Check if it is an anti control message
242  if(!anti_control_message(msg_to_process)) {
243  msg_release(msg_to_process);
244  continue;
245  }
246 
247  break;
248 
249  default:
250  rootsim_error(true, "Received a message which is neither positive nor negative. Aborting...\n");
251  }
252  }
253  }
254 
255  // We have processed all in transit messages.
256  // Actually, during this operation, some new in transit messages could
257  // be placed by other threads. In this case, we loose their presence.
258  // This is not a correctness error. The only issue could be that the
259  // preemptive scheme will not detect this, and some events could
260  // be in fact executed out of order.
261  #ifdef HAVE_PREEMPTION
262  reset_min_in_transit(local_tid);
263  #endif
264 }
