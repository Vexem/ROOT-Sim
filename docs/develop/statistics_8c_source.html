118 static void _rmdir(const char *path) {
119  char buf[MAX_PATHLEN];
120  struct stat st;
121  struct dirent *dirt;
122  DIR *dir;
123 
124  // We could be asked to remove a non-existing folder. In that case, do nothing!
125  if (!(dir = opendir(path))) {
126  return;
127  }
128 
129  while ( (dirt = readdir(dir))) {
130  if ((strcmp(dirt->d_name, ".") == 0) || (strcmp(dirt->d_name, "..") == 0))
131  continue;
132  snprintf(buf, MAX_PATHLEN, "%s/%s", path, dirt->d_name);
133 
134  if (stat(buf, &st) == -1) {
135  rootsim_error(false, "stat() file \"%s\" failed, %s\n", buf, strerror(errno));
136  continue;
137  }
138 
139  if (S_ISLNK(st.st_mode) || S_ISREG(st.st_mode)) {
140  if (unlink(buf) == -1)
141  rootsim_error(false, "unlink() file \"%s\" failed, %s\n", buf, strerror(errno));
142  }
143  else if (S_ISDIR(st.st_mode)) {
144  _rmdir(buf);
145  }
146  }
147  closedir(dir);
148

150  rootsim_error(false, "rmdir() directory \"%s\" failed, %s\n", buf, strerror(errno));

553 static inline void statistics_flush_gvt(double gvt) {
554  register unsigned int i;
555  unsigned int committed = 0;
556  static __thread unsigned int cumulated = 0;
557  double exec_time;
558  char gvt_line[512];
559  int gvt_line_len;
560 
561  // Dump on file only if required
562  if( rootsim_config.stats != STATS_ALL && rootsim_config.stats != STATS_PERF) {
563  return;
564  }
565 
566  exec_time = timer_value_seconds(simulation_timer);
567 
568  // Reduce the committed events from all LPs
569  for(i = 0; i < n_prc_per_thread; i++) {
570  committed += lp_stats_gvt[lid_to_int(LPS_bound(i)->lid)].committed_events;
571  }
572  cumulated += committed;
573 
574  // If we are using a higher level of statistics, buffer data and eventually dump on file
575  if(rootsim_config.stats == STATS_PERF || rootsim_config.stats == STATS_LP || rootsim_config.stats == STATS_ALL) {
576 
577  snprintf(gvt_line, 512, "%f\t%f\t%d\t%d\n", exec_time, gvt, committed, cumulated);
578  gvt_line_len = strlen(gvt_line);
579 
580  if(gvt_buffer.pos + gvt_line_len > GVT_BUFF_SIZE - 2) {
581  statistics_flush_gvt_buffer();
582  }
583 
584  strcat(gvt_buffer.buff, gvt_line);
585  gvt_buffer.pos += gvt_line_len;
586  }
