129 static void _rmdir(const char *path)
130 {
131  char *buf;
132  struct stat st;
133  struct dirent *dirt;
134  DIR *dir;
135 
136  // We could be asked to remove a non-existing folder. In that case, do nothing!
137  if (!(dir = opendir(path))) {
138  return;
139  }
140 
141  while ( (dirt = readdir(dir))) {
142  if ((strcmp(dirt->d_name, ".") == 0) || (strcmp(dirt->d_name, "..") == 0))
143  continue;
144 
145  safe_asprintf(&buf, "%s/%s", path, dirt->d_name);
146 
147  if (stat(buf, &st) == -1) {
148  rootsim_error(false, "stat() file \"%s\" failed, %s\n", buf, strerror(errno));
149  }else if (S_ISLNK(st.st_mode) || S_ISREG(st.st_mode)) {
150  if (unlink(buf) == -1)
151  rootsim_error(false, "unlink() file \"%s\" failed, %s\n", buf, strerror(errno));
152  }
153  else if (S_ISDIR(st.st_mode)) {
154  _rmdir(buf);
155  }
156 
157  rsfree(buf);
158  }
159  closedir(dir);
160 
161  if (rmdir(path) == -1)
162  rootsim_error(false, "rmdir() directory \"%s\" failed, %s\n", path, strerror(errno));
163 }

549 // dump a line on the corresponding statistics file
550 inline void statistics_on_gvt(double gvt)
551 {
552  unsigned int lid;
553  unsigned int committed = 0;
554  static __thread unsigned int cumulated = 0;
555  double exec_time, simtime_advancement, keep_exponential_event_time;
556 
557  // Dump on file only if required
558  if(rootsim_config.stats == STATS_ALL || rootsim_config.stats == STATS_PERF) {
559  exec_time = timer_value_seconds(simulation_timer);
560 
561  // Reduce the committed events from all LPs
562  foreach_bound_lp(lp) {
563  committed += lp_stats_gvt[lp->lid.to_int].committed_events;
564  }
565  cumulated += committed;
566 
567  // fill the row
568  gvt_buf.rows[gvt_buf.pos++] = (struct _gvt_buffer_row_t){exec_time, gvt, committed, cumulated};
569  // check if buffer is full
570  if(gvt_buf.pos >= GVT_BUFF_ROWS){
571  // flush our buffer in the blob file
572  fwrite(gvt_buf.rows, sizeof(struct _gvt_buffer_row_t), gvt_buf.pos, thread_blob_files[local_tid]);
573  gvt_buf.pos = 0;
574  }
575  }
576 
577  thread_stats[local_tid].memory_usage += (double)getCurrentRSS();
578  thread_stats[local_tid].gvt_computations += 1.0;
579 
580  simtime_advancement = gvt - thread_stats[local_tid].gvt_time;
581  if(D_DIFFER_ZERO(thread_stats[local_tid].simtime_advancement)) {
582  // Exponential moving average
583  thread_stats[local_tid].simtime_advancement =
584  0.1 * simtime_advancement +
585  0.9 * thread_stats[local_tid].simtime_advancement;
586  } else {
587  thread_stats[local_tid].simtime_advancement = simtime_advancement;
588  }
589  thread_stats[local_tid].gvt_time = gvt;
590 
591  foreach_bound_lp(lp) {
592  lid = lp->lid.to_int;
593 
594  lp_stats[lid].vec += lp_stats_gvt[lid].vec;
595 
596  lp_stats[lid].exponential_event_time = lp_stats_gvt[lid].exponential_event_time;
597 
598  keep_exponential_event_time = lp_stats_gvt[lid].exponential_event_time;
599  bzero(&lp_stats_gvt[lid], sizeof(struct stat_t));
600  lp_stats_gvt[lid].exponential_event_time = keep_exponential_event_time;
601  }
602 }

638 {

641 
642  // Make output dir if non existant
643  _mkdir(rootsim_config.output_dir);
644 
645  // The whole reduction for the sequential simulation is simply done at the end
646  if(rootsim_config.serial){

648  return;
649  }
650

652  safe_asprintf(&name_buf, "%s/thread_%u_%u/", rootsim_config.output_dir, kid, i);
653  _mkdir(name_buf);
654  rsfree(name_buf);
655  }
656  // create the unique file
657 #ifdef HAVE_MPI
658  if(n_ker > 1) {
659  assign_new_file(unique_files[STAT_FILE_U_NODE], STAT_FILE_NAME_NODE"_%u", kid);
660  if(master_kernel())
661  assign_new_file(unique_files[STAT_FILE_U_GLOBAL], STAT_FILE_NAME_GLOBAL);
662  } else
663 #endif
664  {
665  assign_new_file(unique_files[STAT_FILE_U_NODE], STAT_FILE_NAME_NODE);
666  }
667  // Create files depending on the actual level of verbosity
668  switch(rootsim_config.stats) {
669  case STATS_ALL:
670  case STATS_LP:
671  thread_files[STAT_FILE_T_LP] = rsalloc(sizeof(FILE *) * n_cores);
672  for(i = 0; i < n_cores; ++i) {
673  assign_new_file(thread_files[STAT_FILE_T_LP][i], "thread_%u_%u/%s", kid, i, STAT_FILE_NAME_LP);
674  }
675  if(rootsim_config.stats == STATS_LP) goto stats_lp_jump;
676  /* fall through */
677  case STATS_PERF:
678  thread_files[STAT_FILE_T_GVT] = rsalloc(sizeof(FILE *) * n_cores);
679  thread_blob_files = rsalloc(sizeof(FILE *) * n_cores);
680  for(i = 0; i < n_cores; ++i) {
681  assign_new_file(thread_files[STAT_FILE_T_GVT][i], "thread_%u_%u/%s", kid, i, STAT_FILE_NAME_GVT);
682  assign_new_file(thread_blob_files[i], "thread_%u_%u/.%s_blob", kid, i, STAT_FILE_NAME_GVT);
683  }
684  /* fall through */
685  case STATS_GLOBAL:
686  stats_lp_jump:
687  thread_files[STAT_FILE_T_THREAD] = rsalloc(sizeof(FILE *) * n_cores);
688  for(i = 0; i < n_cores; ++i) {

690  }
691  break;
692  default:

699  lp_stats_gvt = rsalloc(n_prc * sizeof(struct stat_t));

701  thread_stats = rsalloc(n_cores * sizeof(struct stat_t));
